340    def __init__(self, flags: int = IN_CLOEXEC) -> None:
341        """
342        `flags` is a bit set of these values:
343        - `IN_CLOEXEC` - Close inotify file descriptor on exec.
344        - `IN_NONBLOCK` - Open inotify file descriptor as non-blocking.
345
346        It's recommended to pass the `IN_CLOEXEC` flag (default) to close the
347        file descriptor on [exec*(2)](https://linux.die.net/man/3/execl).
348
349        This calls [inotify_init1(2)](https://linux.die.net/man/2/inotify_init1)
350        and thus might raise an `OSError` with one of these `errno` values:
351        - `EINVAL`
352        - `EMFILE`
353        - `ENOMEM`
354        - `ENOSYS` if your libc doesn't support [inotify_init1(2)](https://linux.die.net/man/2/inotify_init1)
355        """
356        self._inotify_fd = -1
357        self._inotify_fd = inotify_init1(flags)
358        self._inotify_stream = os.fdopen(self._inotify_fd, 'rb')
359
360        self._path_to_wd = {}
361        self._wd_to_path = {}

379    def close(self) -> None:
380        """
381        Close the inotify handle.
382
383        Can safely be called multiple times, but you
384        can't call any other methods once closed.
385        """
386        try:
387            if self._inotify_fd != -1:

389                # self._inotify_stream is not assigned, but self._inotify_fd

391                # Since this method is called in __del__() this state needs
392                # be handled.
393                self._inotify_stream.close()
394        finally:
395            self._inotify_fd = -1

406    def add_watch(self, path: str, mask: int = IN_ALL_EVENTS) -> int:
407        """
408        Add a watch path.
409
410        `mask` is a bit set of these event flags:
411        - `IN_ACCESS` - File was accessed.
412        - `IN_MODIFY` - File was modified.
413        - `IN_ATTRIB` - Metadata changed.
414        - `IN_CLOSE_WRITE` - Writtable file was closed.
415        - `IN_CLOSE_NOWRITE` - Unwrittable file closed.
416        - `IN_OPEN` - File was opened.
417        - `IN_MOVED_FROM` - File was moved from X.
418        - `IN_MOVED_TO` - File was moved to Y.
419        - `IN_CREATE` - Subfile was created.
420        - `IN_DELETE` - Subfile was deleted.
421        - `IN_DELETE_SELF` - Self was deleted.
422        - `IN_MOVE_SELF` - Self was moved.
423
424        And these additional flags:
425        - `IN_ONLYDIR` - Only watch the path if it is a directory.
426        - `IN_DONT_FOLLOW` - Don't follow symbolic links.
427        - `IN_EXCL_UNLINK` - Exclude events on unlinked objects.
428        - `IN_MASK_CREATE` - Only create watches.
429        - `IN_MASK_ADD` - Add to the mask of an already existing watch.
430        - `IN_ONESHOT` - Only send event once.
431
432        This calls [inotify_add_watch(2)](https://linux.die.net/man/2/inotify_add_watch)
433        and thus might raise one of these exceptions:
434        - `PermissionError` (`EACCES`)
435        - `FileExistsError` (`EEXISTS`)
436        - `FileNotFoundError` (`ENOENT`)
437        - `NotADirectoryError` (`ENOTDIR`)
438        - `OSError` (`WBADF`, `EFAULT`, `EINVAL`, `ENAMETOOLONG`,
439          `ENOMEM`, `ENOSPC`, `ENOSYS` if your libc doesn't support
440          `inotify_rm_watch()`)
441        """
442        path_bytes = path.encode('UTF-8', 'surrogateescape')
443
444        wd = inotify_add_watch(self._inotify_fd, path_bytes, mask)
445        _check_return(wd, path)
446
447        self._path_to_wd[path] = wd
448        self._wd_to_path[wd] = path
449
450        return wd

452    def remove_watch(self, path: str) -> None:
453        """
454        Remove watch by path.
455
456        Does nothing if the path is not watched.
457
458        This calls [inotify_rm_watch(2)](https://linux.die.net/man/2/inotify_rm_watch)
459        and this might raise an `OSError` with one of these `errno` values:
460        - `EBADF`
461        - `EINVAL`
462        - `ENOSYS` if your libc doesn't support [inotify_rm_watch(2)](https://linux.die.net/man/2/inotify_rm_watch)
463        """
464        wd = self._path_to_wd.get(path)
465        if wd is None:
466            _logger.debug('%s: Path is not watched', path)
467            return
468
469        try:
470            res = inotify_rm_watch(self._inotify_fd, wd)
471            _check_return(res, path)
472        finally:
473            del self._path_to_wd[path]
474            del self._wd_to_path[wd]

518    def read_event(self) -> Optional[InotifyEvent]:
519        """
520        Read a single event. Might return `None` if there is none avaialbe.
521        """
522        stream = self._inotify_stream
523        header_bytes = stream.read(_HEADER_STRUCT.size)
524        if not header_bytes:
525            return None
526
527        wd, mask, cookie, filename_len = _HEADER_STRUCT.unpack(header_bytes)
528
529        if filename_len:
530            filename_bytes = stream.read(filename_len)
531            filename = filename_bytes.rstrip(b'\0').decode('UTF-8', 'surrogateescape')
532        else:
533            filename = None
534
535        wd_to_path = self._wd_to_path
536        watch_path = wd_to_path.get(wd)
537
538        if mask & IN_IGNORED and watch_path is not None:
539            wd_to_path.pop(wd, None)
540            self._path_to_wd.pop(watch_path, None)
541
542        if watch_path is None:
543            _logger.debug('Got inotify event for unknown watch handle: %d, mask: %d, cookie: %d', wd, mask, cookie)
544            return None
545
546        return InotifyEvent(wd, mask, cookie, filename_len, watch_path, filename)

548    def read_events(self, terminal_events: int = IN_Q_OVERFLOW | IN_UNMOUNT) -> list[InotifyEvent]:
549        """
550        Read available events. Might return an empty list if there are none
551        available.
552
553        `terminal_events` is per default: `IN_Q_OVERFLOW` | `IN_UNMOUNT`
554
555        **NOTE:** Don't use this in blocking mode! It will never return.
556
557        Raises `TerminalEventException` if any of the flags in `terminal_events`
558        are set in an event `mask`.
559        """
560        stream = self._inotify_stream
561        wd_to_path = self._wd_to_path
562
563        events: list[InotifyEvent] = []
564
565        while header_bytes := stream.read(_HEADER_STRUCT.size):
566            wd, mask, cookie, filename_len = _HEADER_STRUCT.unpack(header_bytes)
567
568            if filename_len:
569                filename_bytes = stream.read(filename_len)
570                filename = filename_bytes.rstrip(b'\0').decode('UTF-8', 'surrogateescape')
571            else:
572                filename = None
573
574            watch_path = wd_to_path.get(wd)
575
576            if mask & terminal_events:
577                raise TerminalEventException(wd, mask, watch_path, filename or None)
578
579            if mask & IN_IGNORED and watch_path is not None:
580                wd_to_path.pop(wd, None)
581                self._path_to_wd.pop(watch_path, None)
582
583            if watch_path is None:
584                _logger.debug('Got inotify event for unknown watch handle: %d, mask: %d, cookie: %d', wd, mask, cookie)
585            else:
586                events.append(InotifyEvent(wd, mask, cookie, filename_len, watch_path, filename))
587
588        return events

550        Read available events. Might return an empty list if there are none
551        available.
552

562

568            if filename_len:
